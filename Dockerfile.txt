FROM ollama/ollama:latest

# Устанавливаем Python
RUN apt-get update && apt-get install -y python3 python3-pip && rm -rf /var/lib/apt/lists/*

# Копируем файлы
WORKDIR /app
COPY server.py .
COPY worm_veil.html .
COPY requirements.txt .

# Устанавливаем зависимости Python
RUN pip3 install requests

# Скачиваем модель при сборке (4.7GB)
RUN ollama pull huihui_ai/dolphin3-abliterated:8b

# Запускаем Ollama и сервер
CMD ollama serve & sleep 10 && python3 server.py
